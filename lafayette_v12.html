🔗 Conexión a Google Sheets
Sigue estos 5 pasos — solo se hacen una vez:
1
Crea una hoja en Google Sheets
Ve a sheets.google.com → crea una hoja nueva → nómbrala "Lafayette Inventario" → dentro crea 3 pestañas (abajo): Inventarios, Entradas, Config
→ Abrir Google Sheets
2
Abre el editor de scripts
Dentro de tu hoja ve a Extensiones → Apps Script. Se abrirá una ventana nueva.
3
Pega este código en Apps Script
Borra todo lo que hay en la ventana y pega esto:
const SS_NAME = 'Lafayette Inventario';

function doPost(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const data = JSON.parse(e.postData.contents);
    const sheetName = data.tipo === 'inventario' ? 'Inventarios' : 'Entradas';
    const sheet = ss.getSheetByName(sheetName);
    data.registros.forEach(r => {
      if (data.tipo === 'inventario') {
        sheet.appendRow([r.fecha, r.nombre, r.unidad, r.valor, new Date().toLocaleString('es-MX')]);
      } else {
        sheet.appendRow([r.fecha, r.nombre, r.unidad, r.cantidad, r.proveedor||'', new Date().toLocaleString('es-MX')]);
      }
    });
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    const action = e.parameter.action || 'test';
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    if (action === 'test') {
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }
    if (action === 'getAll') {
      const invSheet = ss.getSheetByName('Inventarios');
      const entSheet = ss.getSheetByName('Entradas');
      const invData = invSheet.getDataRange().getValues().slice(1)
        .filter(r => r[0])
        .map(r => ({fecha:r[0], nombre:r[1], unidad:r[2], valor:Number(r[3])}));
      const entData = entSheet.getDataRange().getValues().slice(1)
        .filter(r => r[0])
        .map(r => ({fecha:r[0], nombre:r[1], unidad:r[2], cantidad:Number(r[3]), proveedor:r[4]||''}));
      return ContentService.createTextOutput(JSON.stringify({ok:true, inventarios:invData, entradas:entData})).setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
